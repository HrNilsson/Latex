% Changes the chapter title. It removes the text "Chapter #"
\makeatletter
\def\@makechapterhead#1{%
  \vspace*{10\p@}%spacing before title
  {
  \parindent \z@ \raggedright \normalfont
    \ifnum \c@secnumdepth >\m@ne
        \Huge\bfseries \thechapter \vspace{5pt} %chapter number
    \fi
    \Huge \bfseries #1\par %chapter title
    \vskip 30\p@ %spacing after title
  }}
\makeatother
